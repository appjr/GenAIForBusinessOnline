from flask import Flask, request, jsonify
import sqlite3
import hashlib
import secrets
from typing import Optional, Dict

app = Flask(__name__)

# Database helper functions
def get_db_connection():
    """Create secure database connection"""
    conn = sqlite3.connect('database.db')
    conn.row_factory = sqlite3.Row  # Return dict-like rows
    return conn

def hash_password(password: str, salt: str = None) -> tuple:
    """
    Hash password using SHA-256 with salt
    Returns: (hashed_password, salt)
    """
    if salt is None:
        salt = secrets.token_hex(16)
    
    hashed = hashlib.pbkdf2_hmac(
        'sha256',
        password.encode('utf-8'),
        salt.encode('utf-8'),
        100000  # iterations
    )
    return hashed.hex(), salt

# Secure endpoints
@app.route('/user')
def get_user():
    """Get user by ID with SQL injection protection"""
    user_id = request.args.get('id')
    
    # Input validation
    if not user_id:
        return jsonify({'error': 'User ID required'}), 400
    
    try:
        user_id = int(user_id)  # Validate it's a number
    except ValueError:
        return jsonify({'error': 'Invalid user ID format'}), 400
    
    conn = get_db_connection()
    cursor = conn.cursor()
    
    # Parameterized query prevents SQL injection
    cursor.execute('SELECT id, username, email FROM users WHERE id = ?', (user_id,))
    user = cursor.fetchone()
    conn.close()
    
    if user is None:
        return jsonify({'error': 'User not found'}), 404
    
    # Return only safe fields (no passwords!)
    return jsonify({
        'id': user['id'],
        'username': user['username'],
        'email': user['email']
    })

@app.route('/login', methods=['POST'])
def login():
    """Secure login endpoint"""
    username = request.form.get('username')
    password = request.form.get('password')
    
    # Input validation
    if not username or not password:
        return jsonify({'error': 'Username and password required'}), 400
    
    if len(username) > 100 or len(password) > 100:
        return jsonify({'error': 'Invalid input length'}), 400
    
    conn = get_db_connection()
    cursor = conn.cursor()
    
    # Parameterized query
    cursor.execute(
        'SELECT id, username, password_hash, salt FROM users WHERE username = ?',
        (username,)
    )
    user = cursor.fetchone()
    conn.close()
    
    if user is None:
        # Don't reveal if username exists
        return jsonify({'error': 'Invalid credentials'}), 401
    
    # Verify password
    hashed_input, _ = hash_password(password, user['salt'])
    
    if hashed_input != user['password_hash']:
        return jsonify({'error': 'Invalid credentials'}), 401
    
    # Success - return only safe data
    return jsonify({
        'status': 'success',
        'user': {
            'id': user['id'],
            'username': user['username']
        }
    })

if __name__ == '__main__':
    # Never run with debug=True in production
    app.run(debug=False, host='127.0.0.1')
